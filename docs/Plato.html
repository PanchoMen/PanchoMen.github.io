def initialize(nombre, alimentos)
	@nombre = nombre
	@alimentos = Lista.new()
	alimentos.each{|a| @alimentos.push_back(a)}
	@proteinas = 0
	@carbohidratos = 0
	@lipidos = 0
	alimentos.each do |a|
		@proteinas += a.proteinas
		@carbohidratos += a.carbohidratos
		@lipidos += a.lipidos
	end
end

def porcentaje(param)
	total = ((@proteinas + @carbohidratos + @lipidos) / 100.0)
	case param
	when "proteinas"
		return (@proteinas / total).round(2)
	when "carbohidratos"
		return (@carbohidratos / total).round(2)
	when "lipidos"
                       return (@lipidos / total).round(2)
	else
		puts "ERROR en par√°metro"
	end
end

def to_s
	s = "#{@nombre}:\n"
	@alimentos.each{|a| s+= "\t-#{a.to_s}\n"}
	s += "Proteinas: #{@proteinas}, Carbohidratos: #{@carbohidratos}, Lipidos: #{@lipidos}, Valor Enegergetico: #{vct}\n"
end
