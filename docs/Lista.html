def insert(valor)
   	if (empty?)
     		@inicio = Node.new(valor,nil,nil)
     		@final = @inicio
   	elsif (@size == 1)
     		@inicio = Node.new(valor,@final,nil)
     		@final.prev = @inicio
   	elsif (@size > 1)
			nuevo = Node.new(valor,@inicio.next,nil)
     		ant_inicio = Node.new(@inicio.value,@inicio.next,nuevo)
			@inicio = nuevo
			@inicio.next = ant_inicio
   	end
		@size += 1
end

def IR?(sexo)
	kcal = 0
               prot = 0
               nodo = @inicio
               i = 0
               while (i < @size) do
		kcal += nodo.value.valor_energetico
		prot += nodo.value.proteinas
		nodo = nodo.next
		i += 1
               end

	if(sexo == "Hombre")
		if(kcal > 3000 and prot > 54)
			return true
		else
			return false
		end	
	elsif(sexo == "Mujer")
		if(kcal > 2300 and prot > 41)
                               return true
                       else
                               return false
                       end
	else
		puts "Error en el parámetro sexo"
	end
end

def pop()
   	if(empty?)
      		puts "No hay elementos"
   	elsif(@size == 1)
      		aux = @final.value
      		@size-=1
      		@inicio = Node.new(nil,nil,nil)
      		@final = @inicio
      		aux
   	elsif(@size > 1)
      		aux = @final.value
      		@size-=1
			@final.prev.next = nil
      		@final = @final.prev
      		aux
   	end
end
